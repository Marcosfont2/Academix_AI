    %% Interfaces (Contratos de Comportamento)
    class IProcessadorIA {
        <<interface>>
        +analisarTexto(string texto)
        +gerarPredicao(dados)
    }

    class IVisualizavel {
        <<interface>>
        +getDadosGrafico() JSON
    }

    %% Classes Abstratas (Generalização)
    class Usuario {
        <<abstract>>
        +String nome
        +String email
        +login()
        +logout()
    }

    class EntidadeAcademica {
        <<abstract>>
        +String id
        +String localizacao
        +getMembros()
    }

    %% Classes Reais (Implementações)
    class Estudante {
        +String objetivoAcademico
        +consultarIA(pergunta)
        +verificarViabilidade(meta)
    }

    class Professor {
        +String ID_Lattes
        +List coAutores
        +gerarGrafoColaboracao()
    }

    class Curriculo {
        +List experiencias
        +List publicacoes
        +float indiceProdutividade
        +gerarMapaCalor()
        +calcularMetricas()
    }

    class Instituicao {
        +String nome
        +List linhasPesquisa
        +int rankingImpacto
        +getPoloDeForça()
    }

    class MotorIA {
        +analisarAderencia(curriculo, edital)
        +predizerSucesso(curriculo, meta)
        +recomendarCaminho(perfil)
        +analisarGapCompetencia()
    }

    class Conquista {
        +String titulo
        +DateTime dataGanho
        +exibirBadge()
    }
